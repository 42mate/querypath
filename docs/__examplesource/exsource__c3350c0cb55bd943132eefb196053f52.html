  12. require_once 'src/QueryPath/QueryPath.php';

  14. // Begin with an HTML stub document (XHTML, actually), and navigate to the title.
  15. qp(QueryPath::HTML_STUB, 'title')
  16.   // Add some text to the title
  17.   ->text('Example of QueryPath.')
  18.   // Now look for the <body> element
  19.   ->find(':root body')
  20.   // Inside the body, add a title and paragraph.
  21.   ->append('<h1>This is a test page</h1><p>Test text</p>')
  22.   // Now we select the paragraph we just created inside the body
  23.   ->children('p')
  24.   // Add a 'class="some-class"' attribute to the paragraph
  25.   ->attr('class', 'some-class')
  26.   // And add a style attribute, too, setting the background color.
  27.   ->css('background-color', '#eee')
  28.   // Now go back to the paragraph again
  29.   ->parent()
  30.   // Before the paragraph and the title, add an empty table.
  31.   ->prepend('<table id="my-table"></table>')
  32.   // Now let's go to the table...
  33.   ->find('#my-table')
  34.   // Add a couple of empty rows
  35.   ->append('<tr></tr><tr></tr>')
  36.   // select the rows (both at once)
  37.   ->children()
  38.   // Add a CSS class to both rows
  39.   ->addClass('table-row')
  40.   // Now just get the first row (at position 0)
  41.   ->eq(0)
  42.   // Add a table header in the first row
  43.   ->append('<th>This is the header</th>')
  44.   // Now go to the next row
  45.   ->next()
  46.   // Add some data to this row
  47.   ->append('<td>This is the data</td>')
  48.   // Write it all out as HTML
  49.   ->writeHTML();
  50. ?>
