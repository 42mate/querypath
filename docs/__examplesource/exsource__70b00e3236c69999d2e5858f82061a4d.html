Using XML

  1. <?php
  2. /**
  3.  * Using QueryPath.
  4.  *
  5.  * This file contains an example of how QueryPath can be used
  6.  * to generate XML.
  7.  * @package Examples
  8.  * @author M Butcher <matt@aleph-null.tv>
  9.  * @license LGPL The GNU Lesser GPL (LGPL) or an MIT-like license.
  10.  */
  11.  
  12. require_once '../src/QueryPath/QueryPath.php';
  13.  
  14.  
  15. // Create a new XML document wrapped in a QueryPath.
  16. // By default, it will point to the root element,
  17. // <author/>
  18. $record = qp('<?xml version="1.0"?><author></author>')
  19.   // Add a new last name inside of author.
  20.   ->append('<lastName>Dostoyevsky</lastName>')
  21.   // Select all of the children of <author/>. In this case,
  22.   // that is <lastName/>
  23.   ->children()
  24.   // Oh, wait... we wanted last name to be inside of a <name/> 
  25.   // element. Use wrap to wrap the current element in something:
  26.   ->wrap('<name/>')
  27.   // And before last name, we want to add first name.
  28.   ->before('<firstName/>')
  29.   // Select first name
  30.   ->prev()
  31.   // Set the text of first name
  32.   ->text('Fyodor')
  33.   // And then after first name, add the patronymic
  34.   ->after('<patronymic>Fyodorovich</patronymic>')
  35.   // Now go back to the root element, the top of the document.
  36.   ->top()
  37.   // Add another tag -- origin.
  38.   ->append('<origin>Russia</origin>')
  39.   // turn the QueryPath contents back into a string. Since we are 
  40.   // at the top of the document, the whole document will be converted
  41.   // to a string.
  42.   ->xml();
  43.  
  44. // Print our results.
  45. print $record;
