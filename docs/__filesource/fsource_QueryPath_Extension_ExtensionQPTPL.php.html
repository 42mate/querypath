  22.   /**
  23.    * Apply a template to an object and then insert the results.
  24.    *
  25.    * This takes a template (an arbitrary fragment of XML/HTML) and an object
  26.    * or array and inserts the contents of the object into the template. The
  27.    * template is then appended to all of the nodes in the current list.
  28.    *
  29.    * Note that the data in the object is *not* escaped before it is merged
  30.    * into the template. For that reason, an object can return markup (as
  31.    * long as it is well-formed).
  32.    * 
  33.    * @param mixed $template 
  34.    *   The template. It can be of any of the types that {@link qp()} supports
  35.    *   natively. Typically it is a string of XML/HTML.
  36.    * @param mixed $object 
  37.    *   Either an object or an associative array.
  38.    *   - In the case where the parameter
  39.    *   is an object, this will introspect the object, looking for getters (a la
  40.    *   Java bean behavior). It will then search the document for CSS classes
  41.    *   that match the method name. The function is then executed and its contents
  42.    *   inserted into the document. (If the function returns NULL, nothing is
  43.    *   inserted.)
  44.    *   - In the case where the paramter is an associative array, the function will
  45.    *   look through the template for CSS classes that match the keys of the
  46.    *   array. When an array key is found, the array value is inserted into the
  47.    *   DOM as a child of the currently matched element(s).
  48.    * @param array $options 
  49.    *   The options for this function. Valid options are:
  50.    *   - <None defined yet>
  51.    * @return QueryPath 
  52.    *   Returns a QueryPath object with all of the changes from the template
  53.    *   applied into the QueryPath elements.
  54.    * @see QueryPath::append()
  55.    */
  56.   public function tpl($template, $object, $options = array()) {
  57.     // Handle default options here.
  58.  
  59.     //$tqp = ($template instanceof QueryPath) ? clone $template: qp($template);
  60.     $tqp = qp($template);
  61.     
  62.     if (is_array($object)) $this->tplArray($tqp, $object, $options);
  63.     elseif (is_object($object)) $this->tplObject($tqp, $object, $options);
  64.     
  65.     return $this->qp->append($tqp);
  66.   }
  67.   
  68.   /**
  69.    * Given one template, do substitutions for all objects.
  70.    *
  71.    * Using this method, one template can be populated from a variety of
  72.    * sources. That one template is then appended to the QueryPath object.
  73.    * @see tpl()
  74.    * @param mixed $template 
  75.    *   The template. It can be of any of the types that {@link qp()} supports
  76.    *   natively. Typically it is a string of XML/HTML.
  77.    * @param array $objects 
  78.    *   An indexed array containing a list of objects or arrays (See {@link tpl()})
  79.    *   that will be merged into the template.
  80.    * @param array $options 
  81.    *   An array of options. See {@link tpl()} for a list.
  82.    * @return QueryPath 
  83.    *   Returns the QueryPath object.
  84.    */
  85.   public function tplAll($template, $objects, $options = array()) {
  86.     $tqp = qp($template, ':root');
  87.     foreach ($objects as $object) {
  88.       if (is_array($object)) 
  89.         $tqp = $this->tplArray($tqp, $object, $options);
  90.       elseif (is_object($object)) 
  91.         $tqp = $this->tplObject($tqp, $object, $options);
  92.     }
  93.     return $this->qp->append($tqp);
  94.   }

  96.   protected function tplArray($tqp, $array, $options = array()) {
  97.     foreach ($array as $key => $value) {
  98.       $first = substr($key,0,1);
  99.       
  100.       // We allow classes and IDs if explicit. Otherwise we assume
  101.       // a class.
  102.       if ($first != '.' && $first != '#') $key = '.' . $key;
  103.       // Breaking the find into two steps is faster.
  104.       if ($tqp->top()->find($key)->size() > 0) {
  105.         $tqp->append($value);
  106.       }
  107.     }
  108.     return $tqp->top();
  109.   }
  110.   
  111.   protected function tplObject($tqp, $object, $options = array()) {
  112.     $ref = new ReflectionObject($object);
  113.     $methods = $ref->getMethods();
  114.     foreach ($methods as $method) {
  115.       if (strpos($method->getName(), 'get') === 0) {
  116.         $cssClass = $this->method2class($method->getName());
  117.         if ($tqp->top()->find($cssClass)->size() > 0) {
  118.           $tqp->append($method->invoke($object));
  119.         }
  120.         else {
  121.           // Revert to the find() that found something.
  122.           $tqp->end();
  123.         }
  124.       }
  125.     }
  126.     return $tqp->top();
  127.   }
  128.   
  129.   protected function method2class($mname) {
  130.     return '.' . substr($mname, 3);
  131.   }
